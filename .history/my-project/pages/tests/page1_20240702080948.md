```sql table4
SELECT
    date,
    SUM(duration) AS total_duration,
    COUNT(*) AS num_tests
FROM test_database.test_data1
GROUP BY date
ORDER BY date;
```

```sql table5
SELECT
    date,
    COUNT(*) AS num_tests,
    SUM(duration) AS total_duration,
    AVG(duration) AS avg_duration,
    MIN(duration) AS min_duration,
    MAX(duration) AS max_duration
FROM test_database.test_data1
GROUP BY date
ORDER BY date;
```
