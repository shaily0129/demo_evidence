```sql table5
SELECT
    "Start Date" AS date,
    COUNT(*) AS num_tests,
    SUM("Duration (seconds)") AS total_duration,
    AVG("Duration (seconds)") AS avg_duration,
    MIN("Duration (seconds)") AS min_duration,
    MAX("Duration (seconds)") AS max_duration
FROM booking2.booking2
GROUP BY "Start Date"
ORDER BY "Start Date";
```
